#include <bits/stdc++.h>
using namespace std;
typedef unsigned __int128 type;
inline type read(type &n)
{
    static int f;
    static char c;
    n = 0;
    f = 1;
    c = getchar();
    while (c < '0' || c > '9')
    {
        if (c == '-') f = -1;
        c = getchar();
    }
    while(c >= '0' && c <= '9')
    {
        n = (n << 3) + (n << 1) + (c ^ '0');
        c = getchar();
    }
    n *= f;
}
void write(type n)
{
    if (n / 10 == 0) putchar((n % 10) | '0');
    else
    {
        write(n / 10);
        putchar((n % 10) | '0');
    }
}
type n, m;
int main()
{
    read(n), read(m);
    ++n;
    for(; m >= 64; m -= 64) n <<= 64, n %= 1000000007;
    for(; m > 0; --m) n <<= 1, n %= 1000000007;
    --n;
    write(n);
}
